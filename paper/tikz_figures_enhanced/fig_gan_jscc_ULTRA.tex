    $L_t = \lambda L_G + \alpha L_{\text{MSE}} + \beta L_{\text{LPIPS}}$\\[2pt]

    $L_{\text{MSE}} = \mathbb{E}[\|\hat{\mathbf{x}} - \mathbf{x}\|_2^2]$ \quad {\tiny (保真)}\\
    $L_{\text{LPIPS}} = \mathbb{E}[d(f(\hat{\mathbf{x}}), f(\mathbf{x}))]$ \quad {\tiny (感知)}\\[3pt]
    \textbf{判别器损失:}\\
    $L_D = -\mathbb{E}[\log D(\mathbf{x}, \mathbf{y})] - \mathbb{E}[\log(1 - D(\hat{\mathbf{x}}, \hat{\mathbf{y}}))]$\\[3pt]
    \textbf{超参数:} $\lambda{=}0.15$, $\alpha{=}0.0023$, $\beta{=}1.0$
